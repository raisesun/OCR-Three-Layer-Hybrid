### 1.2 分层设计说明

| 层次 | 职责 | 响应时间 | 核心组件 |
|------|------|:--------:|---------|
| **HTTP API层** | 接收HTTP请求，返回JSON响应 | <0.1s | FastAPI, Uvicorn |
| **请求路由层** | 统一入口，协调各层处理 | <0.1s | OcrService |
| **文档分类层** | 识别文档类型，路由到对应处理层 | <0.5s | DocumentClassifier |
| **规则提取层** | 固定文档的正则提取 | <1s | RuleProcessor |
| **VLM处理层** | 半固定文档的视觉理解 | 10-60s | VlmProcessor |
| **LLM处理层** | 复杂文档的字段提取 | 15-67s | LlmProcessor |
| **结果整合层** | 统一输出格式 | <0.1s | ResultMerger |
| **缓存层** | 缓存处理结果，避免重复计算 | <0.1s | ResultCache |

### 1.3 技术选型依据

| 组件 | 选型 | 理由 |
|------|------|------|
| HTTP框架 | FastAPI | 异步高性能，自动生成API文档 |

## 四、核心模块设计

### 4.1 OCR服务统一入口（OcrService）

```python
# src/core/ocr_service.py

from typing import Dict, Any, Optional
from loguru import logger
import hashlib
from datetime import datetime

from src.core.classifier import DocumentClassifier
from src.core.rule_processor import RuleProcessor
from src.core.vlm_processor import VlmProcessor
from src.core.llm_processor import LlmProcessor
from src.utils.cache import ResultCache
from src.utils.logger import get_logger

class OcrService:
    """OCR服务统一入口"""
    
    def __init__(self, config: dict):
        self.config = config
        self.logger = get_logger("OcrService")
        
        # 初始化各组件
        self.classifier = DocumentClassifier()
        self.rule_processor = RuleProcessor()
        self.vlm_processor = VlmProcessor(config)
        self.llm_processor = LlmProcessor(config)
        self.cache = ResultCache(max_size=1000)
        
        self.logger.info("OCR服务初始化完成")
    
    async def process(self, image_path: str) -> Dict[str, Any]:
        """处理单张图片"""
        start_time = datetime.now()
        
        try:
            # 1. 计算图片hash，检查缓存
            image_hash = self._compute_hash(image_path)
            cached_result = self.cache.get(image_hash)
            if cached_result:
                self.logger.info(f"命中缓存: {image_path}")
                return cached_result
            
            # 2. 文档分类
            doc_type = self.classifier.classify(image_path)
            self.logger.info(f"文档分类结果: {doc_type}")
            
            # 3. 根据文档类型路由到不同处理器
            if doc_type in ["身份证正面", "身份证背面", "结婚证", "发票", "快递单"]:
                # 规则提取层
                result = self.rule_processor.process(image_path, doc_type)
            
            elif doc_type in ["户口本"]:
                # VLM处理层
                result = self.vlm_processor.process(image_path, doc_type)
            
            elif doc_type in ["购房合同", "房产证", "租房合同", "提单", "不动产权证"]:
                # LLM处理层
                result = self.llm_processor.process(image_path, doc_type)
            
            else:
                # 未知文档类型，尝试VLM
                self.logger.warning(f"未知文档类型: {doc_type}，使用VLM处理")
                result = self.vlm_processor.process(image_path, "通用")
            
            # 4. 添加元信息
            result["meta"] = {
                "document_type": doc_type,
                "process_time": (datetime.now() - start_time).total_seconds(),
                "process_method": self._get_process_method(doc_type),
                "timestamp": datetime.now().isoformat(),
            }
            
            # 5. 缓存结果
            self.cache.set(image_hash, result)
            
            self.logger.info(
                f"处理完成: {doc_type}, "
                f"耗时: {result['meta']['process_time']:.2f}s"
            )
            
            return result
        
        except Exception as e:
            self.logger.error(f"处理失败: {image_path}, 错误: {str(e)}")
            raise
    
    def _compute_hash(self, image_path: str) -> str:
        """计算图片hash"""
        with open(image_path, "rb") as f:
            return hashlib.md5(f.read()).hexdigest()
    
    def _get_process_method(self, doc_type: str) -> str:
        """获取处理方法"""
        if doc_type in ["身份证正面", "身份证背面", "结婚证", "发票", "快递单"]:
            return "规则提取"
        elif doc_type in ["户口本"]:
            return "VLM"
        else:
            return "LLM"
```

### 4.2 文档分类器（DocumentClassifier）

```python
# src/core/classifier.py

import re
from typing import Dict, List, Optional
from loguru import logger

from src.engines.ocr_engine import OcrEngine
from src.utils.logger import get_logger

class DocumentClassifier:
    """文档分类器"""
    
    # 文档类型关键词配置
    KEYWORDS = {
        "身份证正面": {
            "keywords": ["姓名", "性别", "民族", "出生", "住址", "公民身份号码"],
            "min_match": 3,
        },
        "身份证背面": {
            "keywords": ["签发机关", "有效期限"],
            "min_match": 2,
        },
        "户口本": {
            "keywords": ["户主", "与户主关系", "姓名", "性别", "民族", "出生日期"],
            "min_match": 3,
        },
        "结婚证": {
            "keywords": ["结婚证", "持证人", "登记日期", "结婚证字号", "民政部"],
            "min_match": 2,
        },
        "房产证": {
            "keywords": ["房屋所有权证", "权利人", "房屋坐落", "建筑面积", "权证字号"],
            "min_match": 2,
        },
        "购房合同": {
            "keywords": ["商品房买卖合同", "出卖人", "买受人", "建筑面积", "总价款"],
            "min_match": 2,
        },
        "发票": {
            "keywords": ["发票", "发票代码", "开票日期", "金额合计"],
            "min_match": 2,
        },
        "快递单": {
            "keywords": ["运单号", "寄件人", "收件人", "地址"],
            "min_match": 2,
        },
    }
    
    def __init__(self):
        self.logger = get_logger("DocumentClassifier")
        self.ocr_engine = OcrEngine()
    
    def classify(self, image_path: str) -> str:
        """分类文档类型"""
        # OCR提取文本
        ocr_result = self.ocr_engine.predict(image_path)
        text = " ".join([item["text"] for item in ocr_result])
        
        # 计算各文档类型的匹配分数
        scores: Dict[str, int] = {}
        for doc_type, config in self.KEYWORDS.items():
            keywords = config["keywords"]
            min_match = config["min_match"]
            
            score = sum(1 for kw in keywords if kw in text)
            scores[doc_type] = score
            
            self.logger.debug(f"文档类型 '{doc_type}' 匹配分数: {score}")
        
        # 获取最高分
        best_type = max(scores, key=scores.get)
        best_score = scores[best_type]
        
        # 检查是否达到最低匹配要求
        min_match = self.KEYWORDS[best_type]["min_match"]
        if best_score >= min_match:
            self.logger.info(f"文档分类结果: {best_type} (分数: {best_score})")
            return best_type
        
        self.logger.warning(f"无法确定文档类型，最高分: {best_type}={best_score}")
        return "未知"
```

### 4.3 规则提取处理器（RuleProcessor）

```python
# src/core/rule_processor.py

from typing import Dict, Any
from loguru import logger

from src.core.processor_base import BaseProcessor
from src.extractors.id_card_extractor import IDCardFrontExtractor, IDCardBackExtractor
from src.extractors.marriage_extractor import MarriageCertExtractor
from src.extractors.invoice_extractor import InvoiceExtractor
from src.extractors.express_extractor import ExpressExtractor
from src.engines.ocr_engine import OcrEngine
from src.utils.logger import get_logger

class RuleProcessor(BaseProcessor):
    """规则提取处理器"""
    
    EXTRACTORS = {
        "身份证正面": IDCardFrontExtractor,
        "身份证背面": IDCardBackExtractor,
        "结婚证": MarriageCertExtractor,
        "发票": InvoiceExtractor,
        "快递单": ExpressExtractor,
    }
    
    def __init__(self):
        self.logger = get_logger("RuleProcessor")
        self.ocr_engine = OcrEngine()
    
    def process(self, image_path: str, doc_type: str) -> Dict[str, Any]:
        """处理文档"""
        # 获取对应的提取器
        extractor_class = self.EXTRACTORS.get(doc_type)
        if not extractor_class:
            raise ValueError(f"不支持的文档类型: {doc_type}")
        
        # OCR提取文本
        ocr_result = self.ocr_engine.predict(image_path)
        text = "\n".join([item["text"] for item in ocr_result])
        
        # 字段提取
        extractor = extractor_class()
        fields = extractor.extract(text)
        
        # 计算置信度
        confidence = self._calculate_confidence(fields, ocr_result)
        
        return {
            "fields": fields,
            "confidence": confidence,
            "ocr_text": text,
        }
    
    def _calculate_confidence(self, fields: Dict, ocr_result: list) -> float:
        """计算置信度"""
        if not fields:
            return 0.0
        
        # 基于OCR置信度和字段提取完整度计算
        avg_ocr_conf = sum(item.get("confidence", 0) for item in ocr_result) / len(ocr_result)
        field_ratio = len(fields) / 10  # 假设最多10个字段
        
        return min(avg_ocr_conf * 0.7 + field_ratio * 0.3, 1.0)
```

            if response.startswith("```json"):
                response = response[7:]
            if response.startswith("```"):
                response = response[3:]
            if response.endswith("```"):
                response = response[:-3]

        except json.JSONDecodeError as e:
            self.logger.warning(f"JSON解析失败: {str(e)}")
            # 尝试正则提取
            import re
            match = re.search(r'\{[^}]+\}', response)
            if match:
                try:
                    return json.loads(match.group())
                except:
                    pass
            
            return {}
    
    def _calculate_confidence(self, fields: Dict) -> float:
        """计算置信度"""
        if not fields:
            return 0.0
        
        # 基于字段完整度计算
        return min(len(fields) / 8, 1.0)  # 假设8个字段

### 4.5 LLM处理器（LlmProcessor）

```python
# src/core/llm_processor.py

from typing import Dict, Any, List
from loguru import logger
from openai import OpenAI

from src.core.processor_base import BaseProcessor
from src.engines.ocr_engine import OcrEngine
from src.engines.embedding_engine import EmbeddingEngine
from src.utils.logger import get_logger

class LlmProcessor(BaseProcessor):
    """LLM处理器"""
    
    KEY_LISTS = {
        "购房合同": [
            "合同编号", "出卖人", "买受人", "房屋坐落",
            "建筑面积", "总价款", "签订日期", "付款方式",
            "交房日期", "违约责任", "争议解决",
        ],
        "房产证": [
            "权利人", "房屋坐落", "建筑面积", "房屋用途",
            "权证字号", "登记时间", "土地用途", "土地使用年限",
        ],
        "租房合同": [
            "出租人", "承租人", "房屋地址", "月租金",
            "租赁期限", "签订日期", "押金", "付款方式",
        ],
        "提单": [
            "提单号", "托运人", "收货人", "船名",
            "起运港", "目的港", "货物描述", "重量",
        ],
        "不动产权证": [
            "权利人", "坐落", "面积", "用途",
            "权利类型", "不动产单元号", "登记时间",
        ],
    }
    
    def __init__(self, config: dict):
        self.config = config
        self.logger = get_logger("LlmProcessor")
        self.ocr_engine = OcrEngine()
        self.embedding_engine = EmbeddingEngine(config)
        
        # Ollama客户端
        self.client = OpenAI(
            base_url=config.get("ollama_base_url", "http://localhost:11434/v1"),
            api_key="ollama",
        )
    
    def process(self, image_path: str, doc_type: str) -> Dict[str, Any]:
        """处理文档"""
        # 1. OCR全文提取
        ocr_result = self.ocr_engine.predict(image_path)
        ocr_text = "\n".join([item["text"] for item in ocr_result])
        
        # 2. 获取字段列表
        key_list = self.KEY_LISTS.get(doc_type, [])
        if not key_list:
            self.logger.warning(f"未知的文档类型: {doc_type}")
            return {"fields": {}, "confidence": 0.0}
        
        # 3. 文本分块 + 向量化
        chunks = self._split_text(ocr_text, chunk_size=500)
        chunk_embeddings = self.embedding_engine.encode(chunks)
        
        # 4. 向量检索 + LLM提取
        fields = {}
        for key in key_list:
            # 检索相关文本块
            key_embedding = self.embedding_engine.encode([key])[0]
            relevant_chunks = self._retrieve_relevant(
                key_embedding, chunks, chunk_embeddings, top_k=3
            )
            
            # LLM提取
            extracted_value = self._extract_with_llm(key, relevant_chunks, doc_type)
            if extracted_value:
                fields[key] = extracted_value
        
        return {
            "fields": fields,
            "confidence": self._calculate_confidence(fields, key_list),
            "ocr_text": ocr_text,
        }
    
    def _split_text(self, text: str, chunk_size: int = 500) -> List[str]:
        """文本分块"""
        chunks = []
        lines = text.split("\n")
        current_chunk = []
        current_size = 0
        
        for line in lines:
            line_size = len(line)
            if current_size + line_size > chunk_size and current_chunk:
                chunks.append("\n".join(current_chunk))
                current_chunk = [line]
                current_size = line_size
            else:
                current_chunk.append(line)
                current_size += line_size
        
        if current_chunk:
            chunks.append("\n".join(current_chunk))
        
        return chunks
    
    def _retrieve_relevant(
        self,
        key_embedding: list,
        chunks: List[str],
        chunk_embeddings: list,
        top_k: int = 3,
    ) -> List[str]:
        """检索相关文本块"""
        import numpy as np
        
        # 计算相似度
        similarities = []
        for chunk_emb in chunk_embeddings:
            sim = np.dot(key_embedding, chunk_emb) / (
                np.linalg.norm(key_embedding) * np.linalg.norm(chunk_emb)
            )
            similarities.append(sim)
        
        # 获取top_k
        top_indices = np.argsort(similarities)[-top_k:][::-1]
        return [chunks[i] for i in top_indices]
    
    def _extract_with_llm(
        self,
        key: str,
        relevant_chunks: List[str],
        doc_type: str,
    ) -> str:
        """使用LLM提取字段"""
        context = "\n---\n".join(relevant_chunks)
        
        prompt = f"""你是一个专业的{doc_type}信息提取助手。

请从以下文本中提取"{key}"字段的值。

文本内容：
---
{context}
---

要求：
1. 只输出字段的值，不要输出其他内容
2. 如果找不到，输出空字符串
3. 尽量简洁准确

{key}的值："""
        
        try:
            response = self.client.chat.completions.create(
                model=self.config.get("llm_model", "qwen3.5:4b"),
                messages=[
                    {"role": "system", "content": "你是专业的文档信息提取助手。"},
                    {"role": "user", "content": prompt},
                ],
                max_tokens=100,
                temperature=0.1,
            )
            
            value = response.choices[0].message.content.strip()
            return value if value else ""
        
        except Exception as e:
            self.logger.error(f"LLM提取失败: {str(e)}")
            return ""
    
    def _calculate_confidence(self, fields: Dict, key_list: List) -> float:
        """计算置信度"""
        if not key_list:
            return 0.0
        
        return len(fields) / len(key_list)
```

            ocr_result.append({
                "text": text,
                "confidence": confidence,
                "box": box,
            })
        
        self.logger.debug(f"OCR结果: {len(ocr_result)} 行文本")

# src/engines/llama_engine.py

import subprocess
import requests
import time
from typing import Optional
from loguru import logger

from src.utils.logger import get_logger

class LlamaEngine:
    """llama.cpp引擎封装"""
    
    def __init__(self, config: dict):
        self.config = config
        self.logger = get_logger("LlamaEngine")
        
        self.model_path = config.get("glm_ocr_model_path")
        self.host = config.get("llama_host", "0.0.0.0")
        self.port = config.get("llama_port", 8080)
        self.base_url = f"http://{self.host}:{self.port}"
        
        self.server_process: Optional[subprocess.Popen] = None
    
    def start_server(self):
        """启动llama服务"""
        if self.server_process and self.server_process.poll() is None:
            self.logger.info("llama服务已在运行")
            return
        
        cmd = [
            "llama-server",
            "-m", self.model_path,
            "--host", self.host,
            "--port", str(self.port),
            "-c", "4096",      # 上下文长度
            "-ngl", "0",       # CPU only
            "--threads", "8",  # 线程数
        ]
        
        self.logger.info(f"启动llama服务: {' '.join(cmd)}")
        self.server_process = subprocess.Popen(
            cmd,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
        )
        
        # 等待服务就绪
        time.sleep(5)
        self.logger.info("llama服务启动完成")
    
    def stop_server(self):
        """停止llama服务"""
        if self.server_process:
            self.server_process.terminate()
            self.server_process.wait()
            self.logger.info("llama服务已停止")
    
    def chat_with_image(
        self,
        image_path: str,
        prompt: str,
        max_tokens: int = 1000,
        temperature: float = 0.1,
    ) -> str:
        """图文对话"""
        # 构造请求
        response = requests.post(
            f"{self.base_url}/v1/chat/completions",
            json={
                "messages": [
                    {
                        "role": "system",
                        "content": "你是专业的文档识别助手。",
                    },
                    {
                        "role": "user",
                        "content": [
                            {"type": "text", "text": prompt},
                            {
                                "type": "image_url",
                                "image_url": {"url": f"file://{image_path}"},
                            },
                        ],
                    },
                ],
                "max_tokens": max_tokens,
                "temperature": temperature,
            },
            timeout=120,
        )
        
        response.raise_for_status()
        
        result = response.json()
        return result["choices"][0]["message"]["content"]

from pathlib import Path

class Settings:
    """配置管理"""
    
    def __init__(self, config_path: str = None):
        if config_path is None:
            config_path = Path(__file__).parent / "config.yaml"
        
        with open(config_path, "r", encoding="utf-8") as f:
            self.config = yaml.safe_load(f)
    
    def get(self, key: str, default: Any = None) -> Any:
        """获取配置项"""
        keys = key.split(".")
        value = self.config
        
        for k in keys:
            if isinstance(value, dict):
                value = value.get(k)
            else:
                return default
            
            if value is None:
                return default
        
        return value
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        return self.config

### 8.2 缓存实现

```python
# src/utils/cache.py

from functools import lru_cache
from typing import Any, Optional
import hashlib

class ResultCache:
    """结果缓存"""
    
    def __init__(self, max_size: int = 1000, ttl: int = 3600):
        self.max_size = max_size
        self.ttl = ttl
        self._cache = {}
        self._timestamps = {}
    
    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        if key in self._cache:
            # 检查是否过期
            import time
            if time.time() - self._timestamps[key] < self.ttl:
                return self._cache[key]
            else:
                # 过期，删除
                del self._cache[key]
                del self._timestamps[key]
        
        return None
    
    def set(self, key: str, value: Any):
        """设置缓存"""
        import time
        
        # 如果缓存满了，删除最旧的
        if len(self._cache) >= self.max_size:
            oldest_key = min(self._timestamps, key=self._timestamps.get)
            del self._cache[oldest_key]
            del self._timestamps[oldest_key]
        
        self._cache[key] = value
        self._timestamps[key] = time.time()
```

### 12.3 常见问题

**Q1: GLM-OCR速度太慢怎么办？**

A: 使用llama.cpp + GGUF Q4量化，预估从338秒降到30-60秒。如果仍然太慢，降级到PaddleOCR-VL-0.9B或PP-ChatOCRv4。

**Q2: Qwen3.5-4B速度超标怎么办？**

A: 降级到Qwen2.5-3B（速度快30-40%），或使用Q3量化（速度快20%），或限制max_tokens=500。

**Q3: 户口本识别准确率低怎么办？**

A: 针对户口本设计专门的规则（表格结构解析），或引入VLM层（GLM-OCR）处理。

**Q4: 如何保证99%准确率？**

A: 1) 固定文档用规则层（100%）；2) 复杂文档优化Prompt工程；3) 引入人工复核环节。
